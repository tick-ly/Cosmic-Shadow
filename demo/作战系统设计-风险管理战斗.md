# 宇宙之影：作战系统设计（风险管理战斗）

## 核心设计理念

**战斗 = 风险管理决策**

不同于传统游戏的"资源管理"，本游戏的战斗核心是"风险管理"。每个超能力使用者都在与物理法则进行危险的交易，成功带来战术优势，失败可能导致灾难性后果。

## 战斗系统架构

### 双层作战结构

```
战略层（大地图）
    ↓ 资源分配、部队调动
战役层（战术地图）
    ↓ 具体战斗执行
单位层（个体战斗）
    ↓ 风险决策、能力使用
```

### 战斗流程

```python
class CombatFlow:
    """战斗流程控制"""

    def __init__(self):
        self.phase = "deployment"
        self.combat_units = []
        self.enemy_forces = []
        self.environment = None
        self.turn_counter = 0

    def execute_combat_turn(self):
        """执行一个战斗回合"""

        # 1. 情报评估阶段
        intelligence = self.gather_intelligence()

        # 2. 决策制定阶段
        decisions = self.make_tactical_decisions(intelligence)

        # 3. 风险评估阶段
        risk_assessments = self.assess_all_risks(decisions)

        # 4. 执行阶段
        results = self.execute_decisions(decisions, risk_assessments)

        # 5. 结果处理阶段
        self.process_combat_results(results)

        # 6. 代价结算阶段
        self.settle_all_costs()
```

### 能力使用决策

```python
class SkillUsageDecision:
    """技能使用决策系统"""

    def evaluate_skill_opportunity(self, unit, skill, combat_context):
        """评估技能使用机会"""

        # 1. 基础可行性检查
        if not self.check_basic_feasibility(unit, skill):
            return {"可行": False, "原因": "基础条件不满足"}

        # 2. 战术价值评估
        tactical_value = self.assess_tactical_value(skill, combat_context)

        # 3. 风险评估
        risk_analysis = self.analyze_risks(unit, skill, combat_context)

        # 4. 代价评估
        cost_analysis = self.analyze_costs(unit, skill)

        # 5. 综合决策
        decision_score = self.calculate_decision_score(
            tactical_value, risk_analysis, cost_analysis
        )

        return {
            "可行": True,
            "战术价值": tactical_value,
            "风险分析": risk_analysis,
            "代价分析": cost_analysis,
            "决策分数": decision_score,
            "建议": self.generate_recommendation(decision_score)
        }

    def analyze_risks(self, unit, skill, context):
        """详细风险分析"""

        risks = {
            "失败风险": {
                "概率": skill.instability_rate,
                "后果": self.analyze_failure_consequences(skill),
                "严重程度": self.assess_failure_severity(skill, unit)
            },
            "反噬风险": {
                "概率": skill.backlash_risk,
                "后果": skill.backlash_effects,
                "严重程度": self.assess_backlash_severity(skill, unit)
            },
            "累积风险": {
                "现实债务": unit.reality_debt + skill.reality_debt,
                "身体负担": unit.body_strain + skill.body_strain,
                "精神消耗": unit.mental_energy - skill.energy_consumption
            }
        }

        # 环境因素修正
        risks = self.apply_environmental_modifiers(risks, context)

        # 单位状态修正
        risks = self.apply_unit_state_modifiers(risks, unit)

        return risks

    def calculate_decision_score(self, tactical_value, risks, costs):
        """计算决策分数"""

        # 战术价值权重
        value_weight = 1.0

        # 风险惩罚
        risk_penalty = (
            risks["失败风险"]["概率"] * 0.3 +
            risks["反噬风险"]["概率"] * 0.5 +
            (risks["累积风险"]["现实债务"] / 10000) * 0.2
        )

        # 代价惩罚
        cost_penalty = (
            costs["精神消耗"] * 0.3 +
            costs["身体负担"] * 0.4 +
            (costs["现实债务"] / 1000) * 0.3
        ) / 100

        # 最终得分
        final_score = (tactical_value * value_weight) - (risk_penalty + cost_penalty)

        return max(0, min(100, final_score))
```

```python
class RiskClassification:

    RISK_LEVELS = {
        "极低": (0, 20),
        "低": (20, 40),
        "中等": (40, 60),
        "高": (60, 80),
```

### 现实债务系统

```python
class RealityDebtSystem:
    """现实债务管理系统"""

    def __init__(self):
        self.debt_thresholds = {
            "warning": 500,      # 警告阈值
            "critical": 1000,    # 危险阈值
            "catastrophic": 2000 # 灾难阈值
        }

    def process_debt_accumulation(self, unit):
        """处理债务累积"""

        current_debt = unit.reality_debt

        if current_debt > self.debt_thresholds["catastrophic"]:
            return self.trigger_catastrophic_event(unit)
        elif current_debt > self.debt_thresholds["critical"]:
            return self.trigger_critical_effects(unit)
        elif current_debt > self.debt_thresholds["warning"]:
            return self.trigger_warning_effects(unit)
        else:
            return {"状态": "正常", "效果": []}

    def trigger_warning_effects(self, unit):
        """触发警告级效果"""

        effects = [
            "现实排斥感增强",
            "轻微的存在不稳定",
            "能力使用难度增加"
        ]

        # 游戏机制影响
        unit.all_skills_instability_increase(5)
        unit.skill_effectiveness_reduce(0.9)

        return {
            "等级": "警告",
            "效果": effects,
            "建议": "减少能力使用，让债务自然消散"
        }

    def trigger_critical_effects(self, unit):
        """触发危险级效果"""

        effects = [

            "存在性抹除",

            "物理法则强制修正",
            "时空错位"

            "等级": "灾难",
            "可能结果": outcomes,
            "紧急措施": "立即寻求专业援助或特殊物品"
        }
```

```python
class CooperativeCombatSystem:
    """协同作战系统"""

    def evaluate_team_combination(self, units):
        """评估队伍组合"""

        team_analysis = {
            "能力覆盖": self.analyze_ability_coverage(units),
            "协同潜力": self.analyze_synergy_potential(units),
            "风险分配": self.analyze_risk_distribution(units),
            "补位能力": self.analyze_backup_capacity(units)
        }

        return team_analysis

    def design_combo_attack(self, primary_unit, supporting_units):
        """设计组合攻击"""

        # 分析主要技能
        primary_skill = primary_unit.preferred_skill

        # 寻找协同技能
        synergies = []
        for supporter in supporting_units:
            for skill in supporter.available_skills:
                synergy_value = self.calculate_synergy(
                    primary_skill, skill
                )
                if synergy_value > 1.2:  # 协同阈值
                    synergies.append({
                        "支援单位": supporter,
                        "支援技能": skill,
                        "协同值": synergy_value,
                        "额外风险": synergy_value * 0.5,
                        "综合评估": synergy_value / (1 + synergy_value * 0.5)
                    })

        # 按综合评估排序
        synergies.sort(key=lambda x: x["综合评估"], reverse=True)

        return synergies[:3]  # 返回前3个最佳组合
```

```python
            "行动": action,

        }

        # 计算综合得分
        evaluation["综合得分"] = self.calculate_action_score(evaluation)

        base_tolerance = self.personality.base_risk_tolerance
```

```python
class CombatResolutionSystem:
    """战斗结算系统"""

    def process_combat_results(self, combat_log):
        """处理战斗结果"""

        results = {
            "单位状态": self.process_unit_status(combat_log),
            "经验收获": self.calculate_experience(combat_log),
            "债务结算": self.settle_reality_debts(combat_log),
            "长期影响": self.assess_long_term_effects(combat_log),
            "奖励分配": self.allocate_rewards(combat_log)
        }

        return results

    def process_unit_status(self, combat_log):
        """处理单位状态"""

        unit_statuses = {}

        for unit_entry in combat_log.unit_entries:
            unit = unit_entry.unit

            # 计算身体损伤
            physical_damage = self.calculate_physical_damage(unit_entry)
            unit.physical_health -= physical_damage

            # 计算精神消耗
            mental_drain = self.calculate_mental_drain(unit_entry)
            unit.mental_energy -= mental_drain

            # 处理能力反噬
            backlash_effects = self.process_backlash(unit_entry)

            # 计算稳定度变化
            stability_loss = self.calculate_stability_loss(unit_entry)
            unit.stability -= stability_loss

            unit_statuses[unit.name] = {
                "身体状态": unit.physical_health,
                "精神状态": unit.mental_energy,
                "稳定度": unit.stability,
                "反噬效果": backlash_effects,
                "需要休息": unit.physical_health < 50 or unit.mental_energy < 50
            }

        return unit_statuses

    def settle_reality_debts(self, combat_log):
        """结算现实债务"""

        debt_settlements = {}

        for unit_entry in combat_log.unit_entries:
            unit = unit_entry.unit

            # 当前债务
            current_debt = unit.reality_debt

            # 自然消散（战斗期间）
            natural_dissipation = min(
                current_debt * 0.1,  # 最多消散10%
                50  # 或最多50点
            )

            # 净债务
            net_debt = current_debt - natural_dissipation

            debt_settlements[unit.name] = {
                "战斗前债务": current_debt,
                "新增债务": unit_entry.debt_accumulated,
                "自然消散": natural_dissipation,
                "净债务": net_debt,
                "消散时间": self.estimate_dissipation_time(net_debt)
            }

        return debt_settlements
```

## 总结

### 系统特色

1. **风险管理核心**：每个决策都涉及明确的风险评估
2. **物理真实感**：基于物理法则的能力系统增强沉浸感
3. **策略深度**：多重因素考虑，避免简单最优解
4. **角色成长**：经验积累影响风险承受和决策质量
5. **战术多样性**：环境、协同、时机等多维度战术选择

### 开发优先级

1. **Phase 1**：基础风险管理框架和单位状态系统
2. **Phase 2**：AI决策系统和战斗流程
3. **Phase 3**：协同作战和高级战术系统
4. **Phase 4**：用户体验优化和平衡调整

### 技术要点
